```
sudo npm install -g appium --unsafe-perm=true --allow-root  
```

export ANDROID_HOME=/Applications/ADT/sdk

export PATH=$PATH:$ANDROID_HOME/bin

export PATH=$PATH:$JAVA_HOME/bin
